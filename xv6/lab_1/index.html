sleep

题目要求我们参考echo.c等代码来书写sleep.c，实现我们的sleep功能。 主要是根据argc判断参数的个数，给出提示，若正常参数，则直接调用sleep函数，使用atoi()函数来让参数变为整数，随后调用即可

#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

int main(int argc, char *argv[])
{
  int exit_state = 0;
  if(argc != 2){
    printf("error: useage: sleep <time>\n");
  }else{
    exit_state = sleep(atoi(argv[1]));
  }
  exit(exit_state == 0? 0:1);  // 给出错误的情况，若调用错误返回1
}

pingpong

翻译题目要求即可，pipe()帮助我们实现两个进程之间的通信，在父进程输入a，在子进程读出并且返回a。


#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

int main()
{
    int p2c[2];
    int c2p[2];

    if(pipe(p2c) == -1){
        fprintf(2,"error in pipe\n"); //stderr 的文件描述符为2
        exit(1);
    }

    if(pipe(c2p) == -1){
        fprintf(2,"error in pipe\n"); //stderr 的文件描述符为2
        exit(1);
    }

    int pid = fork(); //获取进程号
    if(pid == -1){
        fprintf(2,"error in fock\n");
    }else if(pid == 0){
        char buf;
        read(p2c[0],&buf,1);
        printf("%d: received ping\n",getpid());
        write(c2p[1],"a",1);
    }else{
        char buf;
        write(p2c[1],"a",1);
        read(c2p[0],&buf,1);
        printf("%d: received pong\n",getpid());
    }
    exit(0);
}

primes

利用进程来筛选质数，其删选的方式是，从2开始，若2之后的数能够被2整除则说明不是质数，被删除，若是质数就利用pipe写入下一个进程。 每次进程的第一个数一定是质数，也就是我们的评判标志。 每个进程只输出一个数(第一个数) 由于xv6的最大子进程限制，所以我们只需要生成2到35即可 相关示意图在题目之中有 注意要先判断是否可以读取到父进程传来的数，若接受不到就直接退出，否则可能产生pipe() error

#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

void primes(int *pip2);

int main(){
  int pip1[2];
  pipe(pip1);  // 将进程相连

  int pid = fork();
  if(pid < 0){
    fprintf(2,"error in fork()\n");
  }else if(pid == 0){
    primes(pip1);
  }else{
    close(pip1[0]); // 父进程只做写操作，不用读 
    for(int i=2;i<=35;i++){
        write(pip1[1],&i,sizeof(int));   //写入2到35到子进程
    }
    close(pip1[1]);
    wait(0); // 等待子进程
  }
  exit(0);
}

void primes(int *pip1){
    close(pip1[1]); // 关闭pip1的写端

    int pip2[2];
    pipe(pip2);
    int firstNum;
    if(read(pip1[0],&firstNum,sizeof(int))) // 从管道之中读取   输出第一个数
    {
        printf("prime %d\n",firstNum);
    }else{
        exit(0);
    }   // 读不到直接退出，防止不必要的fork
    
    int pid = fork();
    if(pid < 0){
        fprintf(2,"error in fork()\n");
        exit(0);
    }else if(pid == 0){   //若是子进程继续筛选
        primes(pip2);
    }else{
        int num;
        while(read(pip1[0],&num,sizeof(int))){
            if(num % firstNum == 0){
                continue;
            }else{
                write(pip2[1],&num,sizeof(int));   // 看是否能够被整除，能则排除，否则写入下一个进程
            }
        }
        close(pip2[1]);
        wait(0); // 父进程等待子进程关闭
    }
    exit(0);
}

find

参考ls之中的列举方式来进行递归操作，基本逻辑为，如果已经是文件，看是否是我们想要的文件，若是文件夹，就进入文件夹之中，依次搜寻 注意一点，在我们列举当前文件夹之中的文件时，有 . .. 表示文件夹，需要跳过它们。在寻找的时候记录路径，用路径寻找即可。

#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"
#include "kernel/fs.h"

void find(char *path, char* target)
{
  char buf[512], *p;
  int fd;
  struct dirent de;
  struct stat st;

  if((fd = open(path, 0)) < 0){
    fprintf(2, "find: cannot open %s\n", path);
    return;
  }

  if(fstat(fd, &st) < 0){
    fprintf(2, "find: cannot stat %s\n", path);
    close(fd);
    return;
  }
    while(read(fd, &de, sizeof(de)) == sizeof(de)){
      strcpy(buf, path);
      p = buf+strlen(buf);
      *p++ = '/';
      if(de.inum == 0)
        continue;

      if(strcmp(de.name, ".") == 0 || strcmp(de.name, "..") == 0)  // 跳过 . .. 避免无限递归
        continue;

      memmove(p, de.name, DIRSIZ);  // 添加到path
      p[DIRSIZ] = 0;

      if(stat(buf, &st) < 0){
        printf("find: cannot stat %s\n", buf);
        continue;
      }

      switch (st.type)
      {
      case T_FILE:
        if(strcmp(de.name,target) == 0){
            printf("%s/%s\n",path,de.name);
        }
        break;
      case T_DIR:
        find(buf,target);
      default:
        break;
      }
  }
  
  close(fd);
}

int main(int argc, char *argv[])
{
  if(argc != 3){
    printf("usage: find <path> <file>\n");
    exit(0);
  }

  find(argv[1], argv[2]);  // 修正参数引用
  exit(0);
}

xargs

重点是理解我们需要将前面运行的结果当作一个参数传递给后面的程序 比如 echo hello | echo world 结果是 world hello 注意我们只需要传递一个参数，每次遇到回车就执行，并且清除刚才的参数(前面指令的运行结果) 所以我们要做的就是从输入端读取运行结果，整个内容作为一个参数添加到后面的程序argv后面，然后fork（）执行即可

#include "kernel/types.h"
#include "user/user.h"
#include "kernel/param.h"
#define MAX_LEN 100
int main(int argc, char *argv[]) {
	char *command = argv[1];
	char *newarg[MAXARG]; // 创建新的参数
	for(int i=1;i<argc;i++){
		newarg[i-1] = argv[i];
	}
	int current = argc-1;
	char  ch[1024];
	int it=0;
	while(read(0,ch+it,1)){
		if(ch[it] != '\n'){           // 若没有倒带结尾，就移动指针
			it++;
		}else{
			ch[it] = '\0';
			newarg[current]=(char*)malloc(it+1);
			memcpy(newarg[current],ch,it+1);
			newarg[current+1]=0;
			if(fork() == 0){
				exec(command,newarg);
				exit(0);
			}else{
				wait(0);
			}
			free(newarg[current]);
			it=0;
		}
	}
	exit(0);
}
